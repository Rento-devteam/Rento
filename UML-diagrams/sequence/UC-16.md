    sequenceDiagram
    participant System as Система
    participant DealService as Сервис сделок
    participant Database as База данных
    participant PaymentService as Платежный сервис
    participant PaymentGateway as Платежный Шлюз
    participant Bank as Банк-эмитент
    participant NotifyService as Сервис уведомлений
    actor Renter as Арендатор
    actor Landlord as Арендодатель
    participant SupportService as Служба поддержки

    Note over System,DealService: Триггеры:<br/>- Обе стороны подтвердили возврат (UC-15)<br/>- Истечение таймера подтверждения

    System->>DealService: trigger_deposit_return(deal_id, trigger_type)
    
    DealService->>Database: Получение данных сделки
    Database-->>DealService: deal_data (status, hold_id, deposit_amount, dispute_status)
    
    DealService->>Database: Проверка наличия претензий/споров
    
    alt Сделка завершена с претензией / активный спор
        Database-->>DealService: dispute_active = true
        
        DealService->>Database: Получение деталей спора
        Database-->>DealService: dispute_details (status, type, resolution_pending)
        
        DealService-->>System: Автоматический возврат заблокирован
        Note over DealService: Запущен процесс разрешения спора<br/>Возврат залога будет выполнен<br/>по решению службы поддержки
        
        rect rgb(255, 248, 240)
            Note over SupportService,Bank: Альтернативный сценарий: Разрешение спора
            
            SupportService->>DealService: resolve_dispute(deal_id, resolution)
            
            alt Полный возврат залога арендатору
                SupportService->>PaymentService: POST /payments/deposit/release (deal_id, full_amount)
                Note over SupportService: Спор решен в пользу арендатора
            else Частичный возврат (компенсация арендодателю)
                SupportService->>PaymentService: POST /payments/deposit/partial (deal_id, deposit_amount, compensation_amount)
                Note over SupportService: Часть залога удерживается как компенсация
            else Полное удержание (залог передан арендодателю)
                SupportService->>PaymentService: POST /payments/deposit/capture (deal_id, deposit_amount)
                Note over SupportService: Залог полностью перечислен арендодателю
            end
            
            PaymentService->>PaymentGateway: execute_dispute_resolution(hold_id, resolution_type, amounts)
            PaymentGateway->>Bank: перевод средств согласно решению
            Bank-->>PaymentGateway: execution_complete
            PaymentGateway-->>PaymentService: resolution_complete
            PaymentService->>Database: Обновление статуса сделки
            PaymentService->>NotifyService: уведомление сторон о решении
        end
        
    else Сделка завершена без претензий
        Database-->>DealService: dispute_active = false, deal_status = "return_confirmed"
        
        DealService->>Database: Проверка таймера подтверждения
        
        alt Подтверждение от обеих сторон получено
            Note over DealService: Сценарий A: Обе стороны подтвердили возврат
            DealService->>DealService: status = "mutual_confirmed"
        else Истек таймер ожидания (арендодатель не подтвердил)
            Note over DealService: Сценарий B: Автоматический возврат по таймеру
            DealService->>Database: Проверка истечения timeout_period
            Database-->>DealService: timer_expired = true
            DealService->>DealService: status = "auto_confirmed"
            DealService->>NotifyService: notify_landlord_auto_return(deal_id)
            NotifyService-->>Landlord: Уведомление: "Срок подтверждения истек. Залог автоматически возвращен арендатору"
        end
        
        DealService->>DealService: Инициализация процесса возврата залога
        
        DealService->>PaymentService: POST /payments/deposit/release (deal_id, hold_id, deposit_amount)
        
        PaymentService->>PaymentGateway: release_hold(hold_id, deposit_amount)
        Note over PaymentGateway: Запрос на разблокировку (отмену холда)<br/>средств залога на карте арендатора
        
        PaymentGateway->>Bank: отмена холда на сумму залога
        Bank->>Bank: Разблокировка средств
        
        alt Успешная разблокировка
            Bank-->>PaymentGateway: hold_released
            PaymentGateway-->>PaymentService: release_confirmed
            PaymentService-->>DealService: deposit_released
            
            DealService->>Database: Обновление статуса сделки
            Database-->>DealService: status: "completed"
            
            DealService->>Database: Фиксация факта возврата залога
            Database-->>DealService: deposit_return_recorded
            
            DealService->>NotifyService: send_deposit_return_notification(deal_id, deposit_amount)
            
            NotifyService-->>Renter: Push/Email: "Залог в размере X руб возвращен на вашу карту"
            NotifyService-->>Landlord: Push/Email: "Сделка завершена. Залог возвращен арендатору"
            
            DealService-->>System: deposit_return_completed
            
        else Ошибка при разблокировке
            Bank-->>PaymentGateway: release_failed (reason)
            PaymentGateway-->>PaymentService: release_failed
            PaymentService-->>DealService: deposit_release_failed
            
            DealService->>Database: Обновление статуса: "deposit_return_pending"
            DealService->>NotifyService: send_admin_alert(deal_id, "deposit_return_failed")
            NotifyService-->>System: Уведомление администратора
            
            DealService->>DealService: Запуск повторной попытки (retry_schedule)
            Note over DealService: Система повторит попытку<br/>через 1 час, 6 часов, 24 часа
            
            DealService-->>System: deposit_return_retry_scheduled
        end
    end
    
    rect rgb(240, 248, 255)
        Note over DealService,Bank: Требования FR-404
        
        Note over PaymentService: "Средства становятся доступны арендатору в течение 24 часов"<br/>
        Note over PaymentService: Фактически зависит от банковских регламентов,<br/>но статус в системе меняется сразу после подтверждения шлюза
    end
    
    rect rgb(240, 255, 240)
        Note over DealService,Database: Таймер автоматического возврата
        
        DealService->>Database: Установка таймера при подтверждении арендатором
        Note over Database: timeout_period = 24 часа (или настроенное значение)<br/>timer_start = time_of_renter_confirmation
        Database-->>DealService: timer_set
        
        alt Арендодатель подтвердил до истечения таймера
            DealService->>DealService: Отмена таймера, стандартный возврат
        else Таймер истек
            DealService->>DealService: Автоматический запуск возврата залога
            DealService->>Database: Фиксация auto_return = true
        end
    end
    
    rect rgb(255, 248, 240)
        Note over PaymentService,PaymentGateway: Технические детали возврата
        
        PaymentService->>PaymentGateway: API запрос на отмену холда<br/>
        Note over PaymentGateway: Метод: /v1/holds/{hold_id}/release<br/>
        Note over PaymentGateway: Тело: {"amount": deposit_amount, "reason": "deal_completed"}
        
        PaymentGateway-->>PaymentService: Ответ:<br/>- статус: "completed"<br/>- release_id: уникальный идентификатор операции<br/>- timestamp: время подтверждения
    end
